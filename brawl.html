async function fetchBrawlStats() { const vercelUrl = "https://brawl-iota.vercel.app/api/brawl"; try { const response = await fetch(vercelUrl); if (!response.ok) throw new Error('Dossier access denied.'); const data = await response.json(); // Updating your UI elements document.querySelector('.stat-item:nth-child(1) .stat-value').innerText = data.trophies.toLocaleString(); document.querySelector('.stat-item:nth-child(2) .stat-value').innerText = data.highestTrophies.toLocaleString(); document.querySelector('.stat-item:nth-child(3) .stat-value').innerText = data['3vs3Victories'].toLocaleString(); document.querySelector('.stat-item:nth-child(4) .stat-value').innerText = data.club?.name || "NO CLUB"; console.log("Stats Synchronized."); } catch (error) { console.error("Fetch Error:", error); document.querySelector('.stat-item:nth-child(1) .stat-value').innerText = "ERR"; } } window.onload = fetchBrawlStats;
